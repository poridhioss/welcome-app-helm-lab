To access your application:

1. Get the application URL:
{{- if eq .Values.service.type "ClusterIP" }}
   kubectl port-forward svc/{{ .Release.Name }} 8080:{{ .Values.service.port }}
   Then open: http://localhost:8080
{{- else if eq .Values.service.type "NodePort" }}
   export NODE_PORT=$(kubectl get svc {{ .Release.Name }} -o jsonpath='{.spec.ports[0].nodePort}')
   export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
   echo "http://$NODE_IP:$NODE_PORT"
{{- else if eq .Values.service.type "LoadBalancer" }}
   NOTE: It may take a few minutes for the LoadBalancer IP to be available.
   kubectl get svc {{ .Release.Name }} -w
{{- end }}

2. Verify the deployment:
   kubectl get pods -l app={{ .Release.Name }}
   kubectl get svc {{ .Release.Name }}
